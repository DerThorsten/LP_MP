# enumerate right nodes with increasing node number
nr ${node_no} ${min_capacity} ${max_capacity}

...

# now the cost funtion is specified
# enumerate linear assignment costs
p1 ${leftNode1} ${rightNode1 cost}
...
...
...

# enumerate pairwise costs
p2 ${leftNode1} ${rightNode1} ${leftNode2} ${rightNode2} ${cost}
...
...
...

# enumerate triple costs
p3 ${leftNode1} ${rightNode1} ${leftNode2} ${rightNode2} ${leftNode3} ${rightNode3} ${cost}
...
...
...
# attention: left nodes and rightnodes need not be distict in p2 and p3!
